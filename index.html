function topFunction() { document.body.scrollTop = 0; // For Safari document.documentElement.scrollTop = 0; // For Chrome, Firefox, IE and Opera // Set duration of the animation (in milliseconds) const duration = 1000; // Get the current scroll position const start = window.pageYOffset; // Calculate the distance to scroll const distance = -start; let currentTime = 0; // Define the animation function const animateScroll = function() { currentTime += 10; const newPosition = easeInOutQuad(currentTime, start, distance, duration); window.scrollTo(0, newPosition); if (currentTime < duration) { requestAnimationFrame(animateScroll); } }; // Define the easing function (optional) const easeInOutQuad = function (t, b, c, d) { t /= d / 2; if (t < 1) return c / 2 * t * t + b; t--; return -c / 2 * (t * (t - 2) - 1) + b; }; // Call the animation function animateScroll(); }
